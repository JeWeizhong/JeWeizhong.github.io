#!/usr/bin/perl -w
use strict;
use Getopt::Long;
use File::Basename;
use FindBin qw($Bin $Script);


my $USAGE = qq{

Fuction：
        test
Useage:
        perl $0 [Options]
Options:
        -h|help         Print this help
        -i| input         input input file
        -o|output         output output file
Other:

};


my ($help,$input,$output);

GetOptions(    # 冒号为设置参数类型
    "help:s"=>\$help,
    "input:s"=>\$input,
    "output:s"=>\$output
);
die "$USAGE\n" unless ($input && $output);

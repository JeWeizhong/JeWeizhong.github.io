使用标准I/O处理文件

#include <stdio.h>
#include <stdlib.h>    // exit()原型

int main(int argc, char *argv[])
{
    int ch;
    FILE *fp;  // 文件指针
    unsigned long count = 0;
    if (argc != 2)
    {
        printf("Usage: %s filename\n",argv[0]);
        exit(EXIT_FAILURE);
    }
    if ((fp = fopen(argv[1],"r")) == NULL)
    {
        printf("Can't open %s\n",argv[1]);
        exit(EXIT_FAILURE);
    }
    while ((ch = getc(fp)) != EOF)
    {
        putc(ch, stdout); // putchar(ch)相同
        count++;
    }
    fclose(fp);
    printf("File %s hash %lu characters\n", argv[1],count);

    return 0;
}

void append(FILE *source, FILE *dest)
{
    size_t bytes;
    static char temp[BUFSIZE];

    while((bytes = fread(temp,sizeof(char),BUFSIZE,source)) > 0)    //二进制形式处理数据
        fwrite(temp,sizeof(char),bytes, dest);
    return 0;
}

char *s_gets(char *st, int n)
{
    char *ret_val;
    char *find;

    ret_val = fgets(st,n, stdin);    // fgets(类型,长度,文件指针)
    if(ret_val)
    {
        find = strchr(st,'\n');    //查找换行符
        if(find)

            *find = '\0';
        else
            while(getchar() != '\n')
                continue;
    }
    return ret_val;
}
